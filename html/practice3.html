Задание 1: Пузырьковая сортировка

Напишите программу, которая сортирует массив чисел по возрастанию с использованием пузырьковой сортировки. Выведите результат в консоль.

            
                // Пример кода
                let arr = [5, 3, 8, 4, 2];

                for (let i = 0; i < arr.length; i++) {
                    for (let j = 0; j < arr.length - 1; j++) {
                        if (arr[j] > arr[j + 1]) {
                            let temp = arr[j];
                            arr[j] = arr[j + 1];
                            arr[j + 1] = temp;
                        }
                    }
                }

                console.log(arr); // [2, 3, 4, 5, 8]
